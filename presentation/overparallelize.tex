\begin{tikzpicture}[
	scale=0.6,
	adder/.style={
		fill=white
	}
]
	% Initial
	\draw[thick] (0, 0) rectangle ++(4, 4);
	\foreach \x in {0,...,3} {
		\foreach \y in {0,...,3} {
			\draw[dashed, opacity=0.25] (\x, \y) rectangle ++(1, 1);
			\node[adder] at ({\x+0.5}, {\y+0.5}) {\tiny+};
		}
	}
	\foreach \x in {0,...,1} {
		\foreach \y in {0,...,1} {
			\draw[dashed, thick, opacity=0.5] ({\x*2}, {\y*2}) rectangle ++(2, 2);
			\node[adder] (tmp) at ({\x*2+1}, {\y*2+1}) {+};
			\draw[<-] (tmp.center) ++( 0.1,  0.1) -- ++( 0.3,  0.3);
			\draw[<-] (tmp.center) ++( 0.1, -0.1) -- ++( 0.3, -0.3);
			\draw[<-] (tmp.center) ++(-0.1,  0.1) -- ++(-0.3,  0.3);
			\draw[<-] (tmp.center) ++(-0.1, -0.1) -- ++(-0.3, -0.3);
		}
	}
	\node[adder] (tmp) at (2, 2) {\Huge+};
	\draw[<-, thick] (tmp.center) ++( 0.2,  0.2)
		.. controls (3, 2.25) .. ++( 0.8,  0.5);
	\draw[<-, thick] (tmp.center) ++( 0.2, -0.2)
		.. controls (3, 1.75) .. ++( 0.8, -0.5);
	\draw[<-, thick] (tmp.center) ++(-0.2,  0.2)
		.. controls (1, 2.25) .. ++(-0.8,  0.5);
	\draw[<-, thick] (tmp.center) ++(-0.2, -0.2)
		.. controls (1, 1.75) .. ++(-0.8, -0.5);
\end{tikzpicture}
